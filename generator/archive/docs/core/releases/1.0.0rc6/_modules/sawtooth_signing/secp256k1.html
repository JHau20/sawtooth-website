[docs]    def from_wif(wif):
        """Decodes a PrivateKey from a wif-encoded string
        """
        try:
            priv = pybitcointools.encode_privkey(wif, 'hex')
            priv = binascii.unhexlify(priv)
            return Secp256k1PrivateKey(
                secp256k1.PrivateKey(priv, ctx=__CTX__))
        except Exception as e:
            raise ParseError('Unable to parse wif key: {}'.format(e))

[docs]    def sign(self, message, private_key):
        try:
            signature = private_key.secp256k1_private_key.ecdsa_sign(message)
            signature = private_key.secp256k1_private_key \
                .ecdsa_serialize_compact(signature)

            return signature.hex()
        except Exception as e:
            raise SigningError('Unable to sign message: {}'.format(str(e)))



[docs]    def verify(self, signature, message, public_key):
        try:
            sig_bytes = bytes.fromhex(signature)

            sig = public_key.secp256k1_public_key.ecdsa_deserialize_compact(
                sig_bytes)
            return public_key.secp256k1_public_key.ecdsa_verify(message, sig)
        # pylint: disable=broad-except
        except Exception:
            return False
